wc_neutral_first_contact_modifier = {
    compatibility_modifier = {
        who = root
        compatibility_target = $CHARACTER$
        min = -15
        max = 15
        multiplier = 1
    }
    modifier = {
        factor = 1.5
        OR = {
            ai_has_cautious_personality = yes
            probably_superstitious = yes
        }
    }
    modifier = {
        factor = 3
        first_contact_neutral_stance_of_reverse_trigger = { target = $CHARACTER$  }
    }
    # Distance, less likely to care if farther away
    modifier = {
        $CHARACTER$ = { is_landed = yes }
        NOT = { ruler_realm_is_on_same_continent_trigger = { CHARACTER = $CHARACTER$ } }
        add = {
            value = 5
            if = {
                limit = {
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value <= squared_distance_small
                        }
                    }
                }
                add = 20
            }
            if = {
                limit = {
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value > squared_distance_small
                        }
                    }
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value <= squared_distance_medium
                        }
                    }
                }
                add = 10
            }
            if = {
                limit = {
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value > squared_distance_medium
                        }
                    }
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value <= squared_distance_large
                        }
                    }
                }
                subtract = 5
            }
            if = {
                limit = {
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value > squared_distance_large
                        }
                    }
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value <= squared_distance_huge
                        }
                    }
                }
                subtract = 10
            }
            if = {
                limit = {
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value > squared_distance_huge
                        }
                    }
                    capital_province = {
                        squared_distance = {
                            target = $CHARACTER$.capital_province
                            value <= squared_distance_almost_massive
                        }
                    }
                }
                subtract = 20
            }
            else = {
                subtract = 30
            }
        }
    }
}
