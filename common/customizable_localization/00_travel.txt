FormerJob = {
	type = character
	
	text = {
		setup_scope = {
			location = { save_scope_as = location }
			if = {
				limit = {
					location.county.holder = {
						exists = dynasty
						dynasty = {
							any_dynasty_member = {
								count >= 3
							}
						}
						NOR = {
							this = root
							root.house ?= this.house
						}
					}
				}
				location.county.holder = {
					save_scope_as = former_employer
				}
			}
			else_if = {
				limit = {
					is_ai = no
					any_ruler = {
						exists = dynasty
						dynasty = {
							any_dynasty_member = {
								count >= 3
							}
						}
						NOR = {
							this = root
							root.house ?= this.house
						}
						capital_province ?= {
