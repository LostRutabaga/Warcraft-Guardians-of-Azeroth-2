first_contact_letter_flirty_trigger = {
    OR = {
        has_trait = lustful
        has_trait = seducer
        has_trait = rakish
    }
    is_attracted_to_gender_of = root
}

first_contact_letter_educated_trigger = {
    OR = {
        has_trait = lifestyle_mystic
        has_trait = education_learning_3
        has_trait = education_learning_4
        has_trait = education_learning_5
        has_trait = scholar
        learning >= high_skill_rating
    }
}

first_contact_de_isolation_trigger = {
    OR = {
        scope:first_contact_nation ?= flag:suramar
    }
}

# Does root have a friendly stance of the target
first_contact_friendly_stance_of_trigger = {
    has_opinion_modifier = {
        target = $target$
        modifier = good_first_contact_opinion
    }
}
