eternal_suffering_value = {
    value = 0
    switch = {
        trigger = has_inactive_trait
        eternal_suffering_5 = { add = 5 }
        eternal_suffering_4 = { add = 4 }
        eternal_suffering_3 = { add = 3 }
        eternal_suffering_2 = { add = 2 }
        eternal_suffering_1 = { add = 1 }
    }
}

current_eternal_suffering_value = {
    value = 0
    if = {
        limit = {
            exists = scope:current_eternal_suffering_value
        }
        add = scope:current_eternal_suffering_value
    }
}
